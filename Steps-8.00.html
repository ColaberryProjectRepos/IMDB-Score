Create Measures and Calculated Column

       Measure and Column Calculations


   Measures

1. TotalMovies = DISTINCTCOUNT(VW_IMDB_Score[Movie_Title])

2. TotalDirect = DISTINCTCOUNT(VW_IMDB_Score[Director_Name])

3. TotalCountry = DISTINCTCOUNT(VW_IMDB_Score[Country])

4. TotalLang = DISTINCTCOUNT(VW_IMDB_Score[Language])

5. Avg_Score = CALCULATE(AVERAGE(VW_IMDB_Score[IMDB_Score]))

6. DOD_AvgScore = IF(ISFILTERED('VW_IMDB_Score'[Calander]), VAR __PREV_YEAR =         

 CALCULATE(AVERAGE(VW_IMDB_Score[IMDB_Score]), DATEADD('VW_IMDB_Score'

   [Caland er].[Date], -10, YEAR)) RETURN DIVIDE(AVERAGE(VW_IMDB_Score[IMDB_Score])

   - __PREV_YEAR,__PREV_YEAR))

7. ArrDecColor = SWITCH (TRUE (), [DOD_AvgScore] >= 0.0000, UNICHAR (9650),

    [DOD_AvgScore] < 0.0000, UNICHAR (9660))

8. MaxIMDB_Score = MAX(VW_IMDB_Score[IMDB_Score])

9. KPI_Color = SWITCH (TRUE (), VW_IMDB_Score[Avg_Score2] >= 3.0, "#207567", 

   [Avg_Score2]   < 3.0, "#680101")

10.  KPI_DOD_Color = SWITCH (TRUE (), [DOD_AvgScore] >= 0.0000, "#207567", 

    [DOD_AvgScore] < 0.0000, "#680101")

11. Goal = 3.0  

12. MinIMDB_Score = MIN(VW_IMDB_Score[IMDB_Score])
